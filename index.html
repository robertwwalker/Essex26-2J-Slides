library(DT)
Files.df <- data.frame(Files=Files) %>%
  mutate(
    Link=paste0(paste0(
    '<a href=',
    paste0("\"https://robertwwalker.github.io/Essex26-2J-Slides/",
           Files,"\"", 
           sep=""),
    ' > ', sep=""),
    str_remove(Files, "ESSSSDA26-"),
    ' Slides</a>',
    sep="")
    ) %>% 
  filter(startsWith(Files, "ESSSSDA26")) %>%
  mutate(File=paste0(paste0(
    '<a href=',
    paste0("\"https://github.com/robertwwalker/Essex-slides/tree/main/",
           Files,"/\"", 
           sep=""),
    ' > ', sep=""),
    Files,
    '</a>',
    sep="")
    ) %>% 
  mutate(Week = as.numeric(str_remove(Files, "ESSSSDA26-Day"))) %>%
  relocate(Week, .before = Files) %>%
  arrange(Week) %>% select(-Files)
datatable(Files.df, escape = FALSE)
